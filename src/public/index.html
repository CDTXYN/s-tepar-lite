      const reader = new FileReader();

      reader.onload = function(event) {
        try {
          const content = event.target.result;
          const data = JSON.parse(content);

          const results = [];

          const issueCounts = {
            missing: 0,
            mismatch: 0,
            expansion: 0
          };

          for (const key in data) {
            const source = data[key].source || '';
            const target = data[key].target || '';

            const sourcePH = extractPlaceholders(source);
            const targetPH = extractPlaceholders(target);

            const missingPH = sourcePH.filter((p) => !targetPH.includes(p));
            const extraPH = targetPH.filter((p) => !sourcePH.includes(p));

            const expansion = calculateExpansion(source, target).toFixed(1);

            if (!target) {
              issueCounts.missing++;
              results.push({
                key,
                statusLabel: 'Error',
                statusClass: 'error',
                details: ['Missing translation.'],
                suggestions: [`Provide a translation for "${source}".`]
              });
            }
            else if (missingPH.length > 0 || extraPH.length > 0) {
              issueCounts.mismatch++;

              const details = [
                `Placeholder mismatch detected. Missing: [${missingPH.join(', ')}] Extra: [${extraPH.join(', ')}].`
              ];

              const suggestions = [];
              if (missingPH.length > 0) {
                suggestions.push(`Add missing placeholders ${missingPH.join(', ')} to target.`);
              }
              if (extraPH.length > 0) {
                suggestions.push(`Remove extra placeholders ${extraPH.join(', ')} from target.`);
              }

              results.push({
                key,
                statusLabel: 'Error',
                statusClass: 'error',
                details,
                suggestions
              });
            }
            else if (source.length > 10 && expansion > 30) {
              issueCounts.expansion++;
              results.push({
                key,
                statusLabel: 'Warning',
                statusClass: 'warning',
                details: [`Expansion is high at ${expansion}%.`],
                suggestions: ['Shorten translation or adjust UI constraints.']
              });
            }
            else if (sourcePH.length > 0) {
              results.push({
                key,
                statusLabel: 'Warning',
                statusClass: 'warning',
                details: [`Placeholders are present and aligned: ${sourcePH.join(', ')}.`],
                suggestions: ['Keep placeholder formatting consistent across locales.']
              });
            }
            else {
              results.push({
                key,
                statusLabel: 'Success',
                statusClass: 'success',
                details: ['Translation check passed.'],
                suggestions: []
              });
            }
          }

          renderSummary(issueCounts);
          renderResults(results);
        }
        catch (err) {
          renderSummary({ missing: '-', mismatch: '-', expansion: '-' });
          document.getElementById('results').innerHTML =
            '

      };
